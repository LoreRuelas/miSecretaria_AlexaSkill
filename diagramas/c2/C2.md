```plantuml
@startuml
!include https://raw.githubusercontent.com/plantuml-stdlib/C4-PlantUML/master/C4_Container.puml

title C2 – Alexa Skill Containers

Person(user, "User")

System_Boundary(skill, "Alexa Skill (Lambda)") {
    Container(handlerLayer, "Handlers Layer", "Python", "Launch, intents, fallback, help, etc.")
    Container(core, "Facade / Core Logic", "Python", "FAC = get_facade(); Procesa operaciones")
    Container(session, "Session Store", "In-memory", "Atributos de sesión")
}

System_Ext(alexa, "Alexa Platform", "Envía requests")

user --> alexa : Voz
alexa --> handlerLayer : Requests (IntentRequest, LaunchRequest, etc.)

handlerLayer --> session : Lectura/Escritura de atributos
handlerLayer --> core : Lógica principal
core --> handlerLayer : Resultados / Respuestas

@enduml
```
